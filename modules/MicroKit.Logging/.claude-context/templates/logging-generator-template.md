```csharp
using System.Text;
using Microsoft.CodeAnalysis;
using Microsoft.CodeAnalysis.CSharp.Syntax;
using Microsoft.CodeAnalysis.Text;

namespace MicroKit.Logging.Generators;

/// <summary>
/// Generates {description} for types decorated with <see cref="{TriggerAttribute}"/>.
/// </summary>
[Generator]
public sealed class {GeneratorName} : IIncrementalGenerator
{
    /// <inheritdoc />
    public void Initialize(IncrementalGeneratorInitializationContext context)
    {
        // Use ForAttributeWithMetadataName for attribute-triggered generation
        // This is the most efficient incremental pipeline entry point
        var provider = context.SyntaxProvider
            .ForAttributeWithMetadataName(
                fullyQualifiedMetadataName: "MicroKit.Logging.{TriggerAttribute}",
                predicate: static (node, _) => node is ClassDeclarationSyntax,
                transform: static (ctx, ct) => GetTargetInfo(ctx, ct))
            .Where(static info => info is not null);

        context.RegisterSourceOutput(provider, static (ctx, info) =>
        {
            if (info is null) return;

            var source = GenerateSource(info);
            ctx.AddSource($"{info.TypeName}.g.cs", SourceText.From(source, Encoding.UTF8));
        });
    }

    private static TargetInfo? GetTargetInfo(
        GeneratorAttributeSyntaxContext context,
        CancellationToken cancellationToken)
    {
        cancellationToken.ThrowIfCancellationRequested();

        if (context.TargetSymbol is not INamedTypeSymbol typeSymbol)
            return null;

        return new TargetInfo(
            TypeName: typeSymbol.Name,
            Namespace: typeSymbol.ContainingNamespace.ToDisplayString());
    }

    private static string GenerateSource(TargetInfo info)
    {
        // Return deterministic output — same input must always produce same output
        return $$"""
            // <auto-generated/>
            #nullable enable

            namespace {{info.Namespace}};

            partial class {{info.TypeName}}
            {
                // TODO: generated members
            }
            """;
    }

    private sealed record TargetInfo(string TypeName, string Namespace);
}
```
